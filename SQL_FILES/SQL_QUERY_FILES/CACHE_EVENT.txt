CREATE TABLE `cache_event` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `user` char(50) CHARACTER SET utf8mb4 DEFAULT NULL,
  `org` varchar(255) CHARACTER SET utf8mb4 DEFAULT NULL,
  `entity` varchar(255) CHARACTER SET utf8mb4 DEFAULT NULL,
  `sender_id` int(11) DEFAULT NULL,
  `_org_id` int(11) DEFAULT NULL,
  `_entity_id` int(11) DEFAULT NULL,
  `timestamp` int(11) DEFAULT NULL,
  `_date` date DEFAULT NULL,
  `_datehour` int(11) DEFAULT NULL,
  `mac_address` varchar(17) CHARACTER SET utf8mb4 DEFAULT NULL,
  `computer_name` varchar(50) CHARACTER SET utf8mb4 DEFAULT NULL,
  `domain_name` varchar(255) CHARACTER SET utf8mb4 DEFAULT NULL,
  `arch` varchar(30) CHARACTER SET utf8mb4 DEFAULT NULL,
  `nacl_arch` varchar(30) CHARACTER SET utf8mb4 DEFAULT NULL,
  `os` varchar(30) CHARACTER SET utf8mb4 DEFAULT NULL,
  `client_version` varchar(30) CHARACTER SET utf8mb4 DEFAULT NULL,
  `full_event` text CHARACTER SET utf8mb4,
  `created` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
  `sync_batch_id` int(11) NOT NULL DEFAULT '0',
  PRIMARY KEY (`id`),
  KEY `user` (`user`),
  KEY `org` (`org`(191)),
  KEY `entity` (`entity`(191)),
  KEY `sender_id` (`sender_id`),
  KEY `timestamp` (`timestamp`),
  KEY `os` (`os`),
  KEY `_date` (`_date`),
  KEY `_datehour` (`_datehour`)
) ENGINE=InnoDB AUTO_INCREMENT=3833893 DEFAULT CHARSET=utf8;
